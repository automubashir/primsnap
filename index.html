Simple API

JavaScript 
// Simple capture
const dataUrl = await PrimSnap.capture('#element');

// Download as PNG
await PrimSnap.download('#element', 'report.png');

// Preview in modal
await PrimSnap.preview('#element');

// With all options
await PrimSnap.capture('#element', {
  format: 'png',
  scale: 2,
  backgroundColor: '#ffffff',
  usePrintStyles: true,

  // Header/Footer
  showHeader: true,
  header: '<span>My Report</span><span>{date}</span>',
  showFooter: true,
  footer: '<span>Confidential</span><span>Page {page}</span>',

  // Page breaks
  avoidBreakInside: ['.section', '.card'],

  // CSS/JS injection
  injectCSS: '.no-print { display: none; }',
  injectJS: (el) => { el.querySelector('.date').textContent = new Date(); },

  // Callbacks
  onProgress: (p, msg) => console.log(p * 100 + '%')
});
